package com.gildedrose;

class GildedRose {
  Item[] items;

  GildedRose(Item[] items) {
    this.items = items;
  }

  public void updateBrie(Item items) {
    if (items.quality < 50) {
      items.quality += 1;
    }
  }

  public void updateSulfuras(Item items) {
    items.quality = 80;
  }

  public void updateBackstage(Item items) {
    items.quality += 1;

    if (items.sellIn < 11) {
      items.quality += 1;
    }
    if (items.sellIn < 6) {
      items.quality += 1;
    }
    if (items.sellIn < 0) {
      items.quality = 0;
    }
  }

  public void updateConjured(Item items) {
    items.quality -= 2;
    if (items.sellIn < 0) {
      items.quality -= 2;
    }
  }

  public void updateFoo(Item items) {
    items.quality -= 1;
    if (items.sellIn < 0) {
      items.quality -= 1;
    }
  }
  
  public void qualityCheck(Item items) {
    if (items.quality > 50) {
      items.quality = 50;
    }
    if (items.quality < 0) {
      items.quality = 0;
    }
  }
  public void sellin(Item items){
    items.sellIn -= 1;
  }  
  public void switchItem(Item items) {
    switch (items.name) {
      case "Aged Brie":
        sellin(items);
        updateBrie(items);
        qualityCheck(items);
        break;

      case "Backstage passes to a TAFKAL80ETC concert":
        sellin(items);
        updateBackstage(items);
        qualityCheck(items);
        break;

      case "Sulfuras, Hand of Ragnaros":
        updateSulfuras(items);
        break;

      case "Conjured":
        sellin(items);
        updateConjured(items);
        qualityCheck(items);
        break;

      default:
        sellin(items);
        updateFoo(items);
        qualityCheck(items);
        break;
    }
  }

  public void updateQuality() {

    for (Item item : items) {
      switchItem(item);
    }
  }
}
